### --views.py--
```python
def cmpPrice(request,id):
    jd = jdSearch(id,5)# 京东爬虫
    tb = tbSearch(id, 5)# 淘宝爬虫
    amz= amzSearch(id, 5)# 亚马逊爬虫
    dd = ddSearch(id, 5)#当当网爬虫
    result=jd+tb+amz+dd
    context = {'goodslist': result}
    return render(request, 'showingPage/home.html', context)
```
